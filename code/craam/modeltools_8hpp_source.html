245 RMDP robustify(const MDP& mdp, bool allowzeros = false){
246  // construct the result first
247  RMDP rmdp;
248  // iterate over all starting states (at t)
249  for(size_t si : indices(mdp)){
250  const auto& s = mdp[si];
251  auto& newstate = rmdp.create_state(si);
252  for(size_t ai : indices(s)){
253  auto& newaction = newstate.create_action(ai);
254  const Transition& t = s[ai].get_outcome();
255  // iterate over transitions next states (at t+1) and add samples
256  if(allowzeros){
257  numvec probabilities = t.probabilities_vector(mdp.state_count());
258  numvec rewards = t.rewards_vector(mdp.state_count());
259  for(size_t nsi : indices(probabilities)){
260  // create the outcome with the appropriate weight
261  Transition& newoutcome =
262  newaction.create_outcome(newaction.size(),
263  probabilities[nsi]);
264  // adds the single sample for each outcome
265  newoutcome.add_sample(nsi, 1.0, rewards[nsi]);
266  }
267  }
268  else{
269  // only consider non-zero probabilities unless allowzeros is used
270  for(size_t nsi : indices(t)){
271  // create the outcome with the appropriate weight
272  Transition& newoutcome =
273  newaction.create_outcome(newaction.size(),
274  t.get_probabilities()[nsi]);
275  // adds the single sample for each outcome
276  newoutcome.add_sample(t.get_indices()[nsi], 1.0, t.get_rewards()[nsi]);
277  }
278  }
279  }
280  }
281  return rmdp;
282 }}
