166 pair<vector<typename Sim::State>, numvec>
167 simulate_return(Sim& sim, prec_t discount,
168  const function<typename Sim::Action(typename Sim::State&)>& policy,
169  long horizon, long runs, prec_t prob_term=0.0,
170  random_device::result_type seed = random_device{}()){
171 
172  long transitions = 0;
173  // initialize random numbers to be used with random termination
174  default_random_engine generator(seed);
175  uniform_real_distribution<double> distribution(0.0,1.0);
176 
177  // pre-initialize output values
178  vector<typename Sim::State> start_states(runs);
179  numvec returns(runs);
180 
181  for(auto run : range(0l,runs)){
182  typename Sim::State state = sim.init_state();
183  start_states[run] = state;
184 
185  prec_t runreturn = 0;
186  for(auto step : range(0l,horizon)){
187  // check from-state termination conditions
188  if(sim.end_condition(state))
189  break;
190 
191  auto action = policy(state);
192  auto reward_state = sim.transition(state,action);
193 
194  auto reward = reward_state.first;
195  auto nextstate = move(reward_state.second);
196 
197  runreturn += reward * pow(discount, step);
198  state = move(nextstate);
199  // test the termination probability only after at least one transition
200  if( (prob_term > 0.0) && (distribution(generator) <= prob_term) )
201  break;
202  transitions++;
203  };
204  returns[run] = runreturn;
205  }
206  return make_pair(move(start_states), move(returns));
